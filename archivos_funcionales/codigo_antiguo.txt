from pypdf import PdfReader
from pdf2image import convert_from_path
import pytesseract
import re

# ================== CONFIGURACIÓN ==================
pdf_path = "files/Ceci by iScanner.pdf"
# pdf_path = "files/NAEL LAMAN LUZURIAGA 230.89.pdf"
# pdf_path = "files/Ceci by iScanner.pdf"

pytesseract.pytesseract.tesseract_cmd = r'C:\Program Files\Tesseract-OCR\tesseract.exe'

# Configuración OCR más efectiva para números y facturas
custom_config = r'--oem 3 --psm 6 -l spa'

# Regex MEJORADO: más flexible
patron_autorizacion = re.compile(
    r'(?:N[Úú]MERO DE AUTORIZACI[Óó]N|NUMERO DE AUTORIZACION|NÚMERO AUTORIZACIÓN|Autorización)[:\s]*(\d{49}|\d{10,})',
    re.IGNORECASE
)

reader = PdfReader(pdf_path)

print(f"📄 El archivo tiene: {len(reader.pages)} páginas\n")

for i, page in enumerate(reader.pages, start=1):
    print(f"{'=' * 30} PAGINA {i} {'=' * 30}")
    
    texto = (page.extract_text() or "").strip()
    numero_encontrado = None

    # 1. Intentar con texto normal (PDF digital)
    if texto:
        match = patron_autorizacion.search(texto)
        if match:
            numero_encontrado = match.group(1)

    # 2. Si no encontró, aplicar OCR
    if not numero_encontrado:
        print("   ⚠️  Aplicando OCR...")
        try:
            images = convert_from_path(pdf_path, first_page=i, last_page=i, dpi=400)
            if images:
                texto_ocr = pytesseract.image_to_string(images[0], config=custom_config)
                
                # Debug: mostrar parte del texto OCR para ver qué está viendo
                # print("   Texto OCR:", texto_ocr[:500])   # descomenta si quieres debug
                
                match = patron_autorizacion.search(texto_ocr)
                if match:
                    numero_encontrado = match.group(1)
        except Exception as e:
            print(f"   ❌ Error OCR: {e}")

    # Resultado final
    if numero_encontrado:
        print(f"✅ Número de Autorización: {numero_encontrado}")
    else:
        print("❌ No se encontró Número de Autorización en esta página")

    print()

from pypdf import PdfReader
from pdf2image import convert_from_path
import pytesseract
import re
from PIL import Image, ImageEnhance


pdf_path = "files/Ceci by iScanner.pdf"
# pdf_path = "files/NAEL LAMAN LUZURIAGA 230.89.pdf"
# pdf_path = "files/JACQUELINE JOHNSON 2_compressed.pdf"
# pdf_path = "files/KAREN CASTRO 466.99.pdf"
# pdf_path = "files/Edison Molina_demo.pdf"

pytesseract.pytesseract.tesseract_cmd = r'C:\Program Files\Tesseract-OCR\tesseract.exe'
custom_config = r'--oem 3 --psm 6 -l spa'
# Regex muy robusto para Número de Autorización
patron = re.compile(r'(\d{49})', re.IGNORECASE)   # Busca directamente 49 dígitos seguidos
patron_fecha_hora = re.compile(
    r'(?:Fecha y Hora De Autorización|FECHA Y HORA DE AUTORIZACIÓN|Fecha y hora de Autorización|FECHA Y HORA|Fecha y Hora|Fecha y hora)[:\s]*(\d{1,2}[/\.-]\d{1,2}[/\.-]\d{2,4}\s+\d{1,2}:\d{2}(?::\d{2})?)',
    re.IGNORECASE
)

reader = PdfReader(pdf_path)

for i, page in enumerate(reader.pages, start=1):
    print(f"{'='*40} PAGINA {i} {'='*40}")
    
    texto = (page.extract_text() or "").strip()
    
    numero_autorizacion = None
    fecha_hora = None

    # 1. Intentar con texto nativo del PDF
    if texto:
        match_num = patron.search(texto)
        if match_num:
            numero_autorizacion = re.sub(r'\s+', '', match_num.group(1))

        match_fecha = patron_fecha_hora.search(texto)
        if match_fecha:
            fecha_hora = match_fecha.group(1)

    # 2. Si no se encontró algún dato, aplicar OCR
    if not numero_autorizacion or not fecha_hora:
        print("   ⚠️  Aplicando OCR...")
        try:
            images = convert_from_path(pdf_path, first_page=i, last_page=i, dpi=450)
            if images:
                img = images[0]
                # Preprocesamiento para mejorar OCR
                img = img.convert('L')                    # Escala de grises
                img = ImageEnhance.Contrast(img).enhance(2.0)   # Aumentar contraste
                img = ImageEnhance.Sharpness(img).enhance(1.5)  # Aumentar nitidez
                
                texto_ocr = pytesseract.image_to_string(img, config=custom_config)

                # Buscar Número de Autorización en OCR
                if not numero_autorizacion:
                    match_num = patron.search(texto_ocr)
                    if match_num:
                        numero_autorizacion = re.sub(r'\s+', '', match_num.group(1))

                # Buscar Fecha y Hora en OCR
                if not fecha_hora:
                    match_fecha = patron_fecha_hora.search(texto_ocr)
                    if match_fecha:
                        fecha_hora = match_fecha.group(1)

        except Exception as e:
            print(f"   ❌ Error en OCR: {e}")
    if numero_autorizacion:
        print(f"✅ Número de Autorización encontrado: {numero_autorizacion}")
    else:
        print("❌ No se encontró Número de Autorización")
    
    if fecha_hora:
        print(f"✅ Fecha y Hora de Autorización : {fecha_hora}")
    else:
        print("❌ No se encontró Fecha y Hora de Autorización")

    print("\n")
